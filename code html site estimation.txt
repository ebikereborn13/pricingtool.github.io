  <script>
    (function () {
      // URL de ton serveur Python (localhost pour les tests)
      const PYTHON_API_URL = "http://localhost:5000/estimate";

      const form = document.getElementById("ebr-form");
      const yearSelect = document.getElementById("ebr-yearSelect");
      const btn = document.getElementById("ebr-submit-btn");
      const resReprise = document.getElementById("res-reprise");
      const resRevente = document.getElementById("res-revente");
      const errorBox = document.getElementById("ebr-error");
      const emailSection = document.getElementById("ebr-email-section");
      const mailtoBtn = document.getElementById("ebr-mailto-btn");

      let globalResult = { reprise: "", revente: "" };

      // Génération des années
      const currentYear = new Date().getFullYear();
      for (let y = currentYear; y >= 2016; y--) {
        const opt = document.createElement("option");
        opt.value = y; opt.textContent = y;
        yearSelect.appendChild(opt);
      }

      form.addEventListener("submit", async function (e) {
        e.preventDefault();
        errorBox.style.display = "none";
        emailSection.style.display = "none";
        btn.disabled = true;
        btn.textContent = "Calcul dynamique...";
        resReprise.textContent = "⏳";
        resRevente.textContent = "⏳";

        // Préparation des données pour Python
        const payload = {
          TypeVelo: document.getElementById("ebr-type").value,
          Marque: document.getElementById("ebr-marque").value,
          Modele: document.getElementById("ebr-modele").value,
          PrixOrigine: document.getElementById("ebr-prix").value,
          Annee: document.getElementById("ebr-yearSelect").value,
          Kilometrage: document.getElementById("ebr-km").value,
          Motorisation: document.getElementById("ebr-moteur").value,
          CapaciteBatterie: document.getElementById("ebr-batterie").value
        };

        try {
          const response = await fetch(PYTHON_API_URL, {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify(payload)
          });

          if (!response.ok) throw new Error("Réponse serveur incorrecte");

          const result = await response.json();

          if (result && !result.error) {
            globalResult.reprise = result.reprise.toLocaleString() + " €";
            globalResult.revente = result.revente.toLocaleString() + " €";
            
            resReprise.textContent = globalResult.reprise;
            resRevente.textContent = globalResult.revente;
            
            emailSection.style.display = "block";
          } else {
            throw new Error(result.error || "Erreur inconnue");
          }
        } catch (err) {
          console.error(err);
          errorBox.style.display = "block";
          resReprise.textContent = "-- €";
          resRevente.textContent = "-- €";
        } finally {
          btn.disabled = false;
          btn.textContent = "Estimer mes tarifs";
        }
      });

      mailtoBtn.addEventListener("click", function() {
        const destinataire = document.getElementById("ebr-user-email").value;
        if(!destinataire || !destinataire.includes('@')) {
          alert("Merci de saisir une adresse email valide.");
          return;
        }

        const sujet = encodeURIComponent("Estimation Reprise : " + document.getElementById("ebr-marque").value);
        let corpsTexte = `Bonjour,\n\nVoici l'estimation pour votre vélo :\n\n`;
        corpsTexte += `Modèle : ${document.getElementById("ebr-marque").value} ${document.getElementById("ebr-modele").value}\n`;
        corpsTexte += `Kilométrage : ${document.getElementById("ebr-km").value} km\n\n`;
        corpsTexte += `💰 TARIF DE RACHAT : ${globalResult.reprise}\n\nValable 14 jours sous réserve de vérification.`;

        window.location.href = `mailto:${destinataire}?subject=${sujet}&body=${encodeURIComponent(corpsTexte)}`;
      });
    })();
  </script>
